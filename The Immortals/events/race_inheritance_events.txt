# Directs every starting character to race inheritance.
TI_race.0002 = {
	scope = none
	hidden = yes

	immediate = {

		every_living_character = {
			trigger_event = { id = TI_race.0003 }
		}
	}
}

	hidden = yes

	trigger = { NOT = {has_character_flag = char_setup_complete} }

	immediate = {
		hidden_effect = {
			race_heritage_effect = yes
		}
	}
}
